90 /**
91 * @brief Figure out if conversions between types like float -> int or int -> short
92 * need scaling applied
93 * @tparam T2 Other type compared to own type
94 * @return true Scaling needs to be applied
95 * @return false No scaling needs to be applied
96 * @see getConversionScale
97 */
98 template <typename T2>
99 static constexpr bool needsScaling() {
100 if (std::is_floating_point<T>::value && std::is_floating_point<T2>::value) {
101 return false;
102 }
103 return std::numeric_limits<T>::max() != std::numeric_limits<T2>::max();
104 }

106 /**
107 * @brief calculate the scale factor needed between the types
108 * @tparam T1 Source type
109 * @tparam T2 target type
110 * @tparam Ratio Floating point type since the value can be smaller than 1
111 * @return constexpr Ratio
112 */
113 template <typename T1, typename T2, typename Ratio = float>
114 static constexpr Ratio getConversionScale() {
115 const Ratio maxt = std::is_floating_point<T2>::value ?
116 1.0 : std::numeric_limits<T2>::max() - 1;
117 const Ratio maxt2 = std::is_floating_point<T1>::value ?
118 1.0 : std::numeric_limits<T1>::max() - 1;
119 return maxt / maxt2;
120 }

122 /**
123 * @brief Set a single channel from an array
124 * @param samples An Array containing the audio samples
125 * @param length Amount of sample to copy
126 * @param channel Channel index
127 * @param offsetDst The offset for the destination buffer (this)
128 */
129 template <typename T2>
130 void set(
131 const T2* samples,
132 Size length,
133 const uchar channel = 0,
134 const Size offsetDst = 0
135 ) {
136 static_assert(std::is_arithmetic<T2>::value, "Need arithmetic type.");
137 if (mChannels <= channel) { return; }
138 TKLB_ASSERT(size() >= offsetDst)
139 length = std::min(length, size() - offsetDst);
140 T* out = get(channel);
141 if (std::is_same<T2, T>::value) {
142 // Types are identical
143 memory::copy(out + offsetDst, samples, sizeof(T) * length);
144 } else {
145 if (!needsScaling<T2>()) {
146 // Both floats means they need to be converted but not scaled
147 for (Size i = 0; i < length; i++) {
148 out[i + offsetDst] = static_cast<T>(samples[i]);
149 }
150 } else {
151 // We also need to scale
152 // float -> int, int -> float, short -> int
153 constexpr auto scale = getConversionScale<T2, T>();
154 for (Size i = 0; i < length; i++) {
155 out[i + offsetDst] = static_cast<T>(samples[i] * scale);
156 }
157 }
158 }
159 }

269 /**
270 * @brief ! Will not keep the contents! Resizes the buffer to the desired length and channel count.
271 * @details If the validSize = 0 it will be set to the new size for convenience.
272 * The size reported back when calling size is can be larger if the data is aligned
273 * @param length The desired length in Samples. 0 will deallocate.
274 * @param channels Desired channel count. 0 will deallocate.
275 */
276 bool resize(const Size length, uchar channels) {
277 if (channels == mChannels && size() == length) { return true; }
278 // We need to ensure each channel is aligned so we add some padding after each channel
279 const auto elementAlign = mBuffer.closestChunkSize(length, mBuffer.Alignment / sizeof(T));
280 mBuffer.resize(0); // deallocate so we don't copy old misaligned signal over
281 mBuffer.resize(channels * elementAlign);
282
283 mChannels = channels;
284
285 if (mValidSize == 0) {
286 mValidSize = length;
287 } else {
288 mValidSize = std::min(mValidSize, length);
289 }
290 return true;
291 }

311 * @brief Add the provided buffer
312 * @param buffer Source buffer object

314 * @param offsetSrc Start offset in the source buffer
315 * @param offsetDst Start offset in the target buffer

317 template <typename T2, class STORAGE2>
318 void add(
319 const AudioBufferTpl<T2, STORAGE2>& buffer,
320 Size length = 0,
321 Size offsetSrc = 0,
322 Size offsetDst = 0
323 ) {
324 TKLB_ASSERT(validSize() >= offsetDst)
325 TKLB_ASSERT(buffer.validSize() >= offsetSrc)
326 length = length == 0 ? buffer.validSize() - offsetSrc : length;
327 length = std::min(buffer.validSize() - offsetSrc, validSize() - offsetDst);
328 const uchar channelCount = std::min(buffer.channels(), channels());
329
330 #ifndef TKLB_NO_SIMD
331 if (std::is_same<T2, T>::value) {
332 const Size vectorize = length - (length % Stride);
333 for (uchar c = 0; c < channelCount; c++) {
334 T* out = get(c) + offsetDst;
335 const T* in = reinterpret_cast<const T*>(buffer[c]) + offsetSrc;
336 for(Size i = 0; i < vectorize; i += Stride) {
337 xsimd::simd_type<T> a = xsimd::load_aligned(in + i);
338 xsimd::simd_type<T> b = xsimd::load_aligned(out + i);
339 xsimd::store_aligned(out + i, (a + b));
340 }
341 for(Size i = vectorize; i < length; i++) {
342 out[i] += in[i];
343 }
344 }
345 return;
346 }
347 #endif

566 * @brief Fill the provided array with the contents of this buffer
567 * @param target The arry to fill
568 * @param channel Which source channel to use
569 * @param length The length of the output
570 * @param offset Start offset in the source (this) buffer
571 * @return The amount of samples written, might be less than requested
572 */
573 template <typename T2>
574 Size put(
575 T2* target,
576 Size length = 0,
577 const uchar channel = 0,
578 const Size offset = 0
579 ) const {
580 static_assert(std::is_arithmetic<T2>::value, "Need arithmetic type.");
581 if (mChannels <= channel) { return 0; }
582 const Size valid = validSize();
583 TKLB_ASSERT(offset <= valid)
584 length = length == 0 ? valid : length;
585 length = std::min(length, valid - offset);
586 const T2* source = get(channel) + offset;
587 if (std::is_same<T2, T>::value) {
588 memory::copy(target, source, sizeof(T) * length);
589 } else {
590 if (!needsScaling<T2>()) {
591 for (Size i = 0; i < length; i++) {
592 target[i] = T2(source[i]);
593 }
594 } else {
595 constexpr auto scale = getConversionScale<T, T2>();
596 for (Size i = 0; i < length; i++) {
597 target[i] = T2(source[i] * scale);
598 }
599 }
600 }

627 * @brief Puts the interleaved contents in the target buffer
628 * @param buffer The array to fill with interleaved audio
629 * @param length Number of frames to interleave (not the length of the interleaved buffer)
630 * @param offset Offset for the sourcebuffer (this)
631 * @return Number of frames emitted
632 */
633 template <typename T2>
634 Size putInterleaved(
635 T2* buffer,
636 Size length = 0,
637 const Size offset = 0
638 ) const {
639 static_assert(std::is_arithmetic<T2>::value, "Need arithmetic type.");
640 const Size valid = validSize();
641 TKLB_ASSERT(offset <= valid)
642 const uchar chan = channels();

644 length = std::min(valid - offset, length);
645 Size out = 0;
646 for (uchar c = 0; c < chan; c++) {
647 Size j = c;
648 const T* data = get(c);
649 if (!needsScaling<T2>()) {
650 for (Size i = 0; i < length; i++) {

656 for (Size i = 0; i < length; i++) {
657 buffer[j] = T2(data[i + offset] * scale);
658 j += chan;
659 }
